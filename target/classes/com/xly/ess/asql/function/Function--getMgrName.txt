DELIMITER $$

USE `xly`$$

DROP FUNCTION IF EXISTS `getMgrName`$$

CREATE DEFINER=`root`@`%` FUNCTION `getMgrName`(userName VARCHAR(10) ) RETURNS VARCHAR(10) CHARSET utf8
BEGIN
    DECLARE  realName VARCHAR(10);
	SELECT  u.`CNAME_` INTO realName FROM bdf2_user u WHERE u.`USERNAME_`=userName;
	RETURN realName;
    END$$

DELIMITER ;
